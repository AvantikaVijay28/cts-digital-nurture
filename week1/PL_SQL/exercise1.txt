DECLARE
    CURSOR c IS
        SELECT CustomerID
        FROM Customers
        WHERE Age > 60;

    cid Customers.CustomerID%TYPE;
BEGIN
    OPEN c;

    LOOP
        FETCH c INTO cid;
        EXIT WHEN c%NOTFOUND;

        UPDATE Loans
        SET InterestRate = InterestRate - 1
        WHERE CustomerID = cid;
    END LOOP;

    CLOSE c;
    COMMIT;
END;

DECLARE
    CURSOR c IS
        SELECT CustomerID
        FROM Customers
        WHERE Balance > 10000;

    cid Customers.CustomerID%TYPE;
BEGIN
    OPEN c;

    LOOP
        FETCH c INTO cid;
        EXIT WHEN c%NOTFOUND;

        UPDATE Customers
        SET IsVIP = 'TRUE'
        WHERE CustomerID = cid;
    END LOOP;

    CLOSE c;
    COMMIT;
END;

3)
SET SERVEROUTPUT ON;

DECLARE
    CURSOR c IS
        SELECT CustomerID, DueDate
        FROM Loans
        WHERE DueDate <= SYSDATE + 30;

    cid Loans.CustomerID%TYPE;
    ddate Loans.DueDate%TYPE;
BEGIN
    OPEN c;

    LOOP
        FETCH c INTO cid, ddate;
        EXIT WHEN c%NOTFOUND;

        DBMS_OUTPUT.PUT_LINE(
            'Reminder for Customer ' || cid
        );
    END LOOP;

    CLOSE c;
END;
